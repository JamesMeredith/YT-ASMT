    // ========== 路由 ==========
    const pageTitles = {
      dashboard: '工作台', faults: '工单管理', devices: '设备管理',
      inspections: '巡检管理', maintenance: '维保台账', parts: '配件管理',

    let currentPage = 'dashboard';
    let dicts = {};

    async function initApp() {
      // 检查登录状态
      if (!API.token) { showLogin(); return; }
      try {
        const r = await API.get('/api/auth/me', { silent:true });
        if (!r) { showLogin(); return; }
        API.user = r.user;
        showApp();
        await loadDicts();
        navigateTo('dashboard');
        startSync();
      } catch(e) { showLogin(); }
    }

    function showLogin() {
      document.getElementById('loginPage').style.display = 'flex';
      document.getElementById('app').style.display = 'none';
      document.getElementById('loginUser').focus();
    }

    function showApp() {
      document.getElementById('loginPage').style.display = 'none';
      document.getElementById('app').style.display = 'flex';
      const u = API.user;
      document.getElementById('userName').textContent = u.real_name || u.username;
      document.getElementById('userAvatar').textContent = (u.real_name || u.username).slice(0,1);
      // 总部/省代可见：系统管理分组
      const navHQ = document.querySelectorAll('.nav-hq-only');
      navHQ.forEach(el => el.style.display = (u.role === 'headquarters' || u.role === 'provincial_agent') ? 'block' : 'none');
      // 售中管理：工程师+代理商可见，总部不可见（总部管理节点定义，不执行项目）
      const navPS = document.querySelectorAll('.nav-pre-sales');
      navPS.forEach(el => el.style.display = (u.role !== 'headquarters') ? 'block' : 'none');
      // 主题
      applyTheme();
    }

    function toggleNavGroup(el) {
      el.parentElement.classList.toggle('collapsed');
    }

    async function doLogin() {
      const u = document.getElementById('loginUser').value.trim();
      const p = document.getElementById('loginPass').value;
      if (!u || !p) return showLoginError('请输入用户名和密码');
      try {
        await API.login(u, p);
        showApp();
        await loadDicts();
        navigateTo('dashboard');
        startSync();
        Toast.success('登录成功');
      } catch(e) {
        showLoginError(e.message || '登录失败');
      }
    }

    function showLoginError(msg) {
      const el = document.getElementById('loginError');
      el.textContent = msg;
      el.style.display = 'block';
    }

    document.getElementById('loginPass').addEventListener('keyup', e => { if (e.key === 'Enter') doLogin(); });

    function logout() {
      API.logout();
      showLogin();
    }

    // ========== 导航 ==========
    function navigateTo(page, params = {}) {
      currentPage = page;
      document.getElementById('pageTitle').textContent = pageTitles[page] || '';
      // 高亮
      document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
      const navItem = document.querySelector(`.nav-item[data-page="${page}"]`);
      if (navItem) navItem.classList.add('active');
      // 渲染
      renderPage(page, params);
    }

    function renderPage(page, params = {}) {
      const c = document.getElementById('pageContent');
      c.innerHTML = '<div class="loading"><span class="spinner"></span>加载中...</div>';
      setTimeout(() => {
        try {
          switch(page) {
            case 'dashboard': renderDashboard(c); break;
            case 'faults': renderFaultList(c, params); break;
            case 'faultDetail': renderFaultDetail(c, params); break;
            case 'devices': renderDeviceList(c, params); break;
            case 'deviceDetail': renderDeviceDetail(c, params); break;
            case 'inspections': renderInspectionList(c); break;
            case 'maintenance': renderMaintenanceList(c); break;
            case 'parts': renderPartsList(c, params); break;
            case 'demands': renderDemandList(c, params); break;
            case 'demandDetail': renderDemandDetail(c, params); break;
            case 'knowledge': renderKnowledgeList(c, params); break;
            case 'knowledgeDetail': renderKnowledgeDetail(c, params); break;
            case 'statistics': renderStatistics(c); break;
            case 'notifications': renderNotifications(c); break;
            case 'users': renderUsersList(c, params); break;
      case 'preSales': renderPreSalesList(c, params); break;
      case 'preSalesDetail': renderPreSalesDetail(c, params); break;
      case 'preSalesCreate': renderPreSalesCreate(c); break;
      case 'preSalesConfig': renderPreSalesNodeConfig(c); break;
      case 'deviceModels': renderDeviceModelsList(c); break;
            case 'profile': renderProfile(c); break;
            default: renderDashboard(c);
          }
        } catch(e) {
          c.innerHTML = `<div class="empty-state"><p>加载失败: ${e.message}</p></div>`;
        }
      }, 50);
    }

    function toggleSidebar() {
      const sb = document.getElementById('sidebar');
      sb.style.width = sb.style.width === '60px' ? '220px' : '60px';
    }

    // ========== 主题 ==========
    function applyTheme() {
      const h = new Date().getHours();
      const auto = (h >= 6 && h < 18) ? 'light' : 'dark';
      const saved = localStorage.getItem('theme');
      document.documentElement.setAttribute('data-theme', saved || auto);
    }

    function toggleTheme() {
      const cur = document.documentElement.getAttribute('data-theme');
      const next = cur === 'dark' ? 'light' : 'dark';
      document.documentElement.setAttribute('data-theme', next);
      localStorage.setItem('theme', next);
    }

      el.textContent = msg;
      c.appendChild(el);
      setTimeout(() => { el.style.opacity='0'; el.style.transition='opacity .3s'; setTimeout(()=>el.remove(),300); }, 3000);
    }

    async function loadPendingFaults() {
      try {
        const r = await API.faults.list({ page:1, page_size:5 });
        const el = document.getElementById('pendingFaultsList');
        if (!r.data || r.data.length === 0) {
          el.innerHTML = '<div class="empty-state"><p>暂无待办工单</p></div>';
          return;
        }
        el.innerHTML = `
          <table class="data-table">
            <thead><tr><th>工单号</th><th>设备编码</th><th>等级</th><th>状态</th><th>描述</th><th>操作</th></tr></thead>
            <tbody>
              ${r.data.map(f => `
                <tr>
                  <td><a href="#" onclick="navigateTo('faultDetail',{fault_no:'${f.fault_no}'});return false;" style="color:var(--primary);">${f.fault_no}</a></td>
                  <td>${f.device_code}</td>
                  <td>${getLevelTag(f.fault_level)}</td>
                  <td>${getStatusTag(f.status)}</td>
                  <td>${f.description.slice(0,30)}${f.description.length>30?'...':''}</td>
                  <td><button class="btn btn-sm btn-primary" onclick="navigateTo('faultDetail',{fault_no:'${f.fault_no}'});">查看</button></td>

            </tbody>
          </table>
        `;
      } catch(e) {
        document.getElementById('pendingFaultsList').innerHTML = '<p style="color:var(--danger)">加载失败</p>';
      }
    }

    function renderSimpleBarChart(data) {
      if (!data || data.length === 0) return '<div class="empty-state"><p>暂无数据</p></div>';
      const max = Math.max(...data.map(d => d.cnt), 1);
      return `
        <div class="chart-bar">
          ${data.map(d => `<div class="chart-bar-item" style="height:${(d.cnt/max)*140}px;" title="${d.date}: ${d.cnt}单"></div>`).join('')}
        </div>
        <div class="chart-bar-label">
          ${data.map(d => `<span>${d.date.slice(5)}</span>`).join('')}
        </div>
      `;
    }

    async function reloadFaultList(page = 1) {

        page,
        page_size: 20,

        level: document.getElementById('faultLevelFilter')?.value || '',
        province: document.getElementById('faultProvinceFilter')?.value || '',

        const r = await API.faults.list(params);
        const wrap = document.getElementById('faultTableWrap');
        if (!r.data || r.data.length === 0) {
          wrap.innerHTML = '<div class="empty-state"><p>暂无工单数据</p></div>';
          return;

    // ========== 工单操作 ==========
    async function doInvestigate(fault_no) {
      const investigation_result = document.getElementById('investResult')?.value;
      const root_cause = document.getElementById('rootCause')?.value;
      if (!root_cause || root_cause.trim().length < 1) return showToast('请输入根因分析', 'error');
      try {
        await API.faults.investigate(fault_no, { investigation_result, root_cause });
        showToast('排查结果已提交', 'success');
        navigateTo('faultDetail', { fault_no });
      } catch(e) { showToast(e.message, 'error'); }
    }

    async function doFix(fault_no) {
      const solution = document.getElementById('solution')?.value;
      if (!solution || solution.length < 20) return showToast('解决方案至少20个字', 'error');
      try {
        await API.faults.fix(fault_no, { solution });
        showToast('修复方案已提交，等待复核', 'success');
        navigateTo('faultDetail', { fault_no });
      } catch(e) { showToast(e.message, 'error'); }
    }

    async function doReview(fault_no) {
      const review_result = document.getElementById('reviewResult')?.value;
      const review_note = document.getElementById('reviewNote')?.value;
      if (!confirm(`确认复核结果：${review_result}？`)) return;
      try {
        await API.faults.review(fault_no, { review_result, review_note });
        showToast('复核完成', 'success');
        navigateTo('faultDetail', { fault_no });
      } catch(e) { showToast(e.message, 'error'); }
    }

    async function doFeedback(fault_no) {
      const feedback_method = document.getElementById('feedbackMethod')?.value;
      const score_1 = parseInt(document.getElementById('score1')?.value || 0);
      const score_2 = parseInt(document.getElementById('score2')?.value || 0);
      const satisfaction = document.getElementById('satisfaction')?.value;
      const feedback_note = document.getElementById('feedbackNote')?.value;
      if (satisfaction === '不满意' && (!feedback_note || feedback_note.trim().length < 1)) return showToast('请填写不满意原因', 'error');
      try {
        await API.faults.feedback(fault_no, { feedback_method, score_1, score_2, score_3: 0, satisfaction, feedback_note });
        showToast('回访已提交', 'success');
        navigateTo('faultDetail', { fault_no });
      } catch(e) { showToast(e.message, 'error'); }
    }

    async function doCreateFault() {
      const device_code = document.getElementById('newFaultDeviceCode')?.value.trim();
      const fault_category_l1 = document.getElementById('newFaultCat1')?.value;
      const description = document.getElementById('newFaultDesc')?.value.trim();
      const contact_person = document.getElementById('newFaultContact')?.value.trim();
      const contact_phone = document.getElementById('newFaultPhone')?.value.trim();
      if (!device_code || !fault_category_l1 || !description || !contact_person || !contact_phone) return showToast('请填写所有必填项', 'error');
      if (description.length < 10) return showToast('故障描述至少10个字', 'error');
      if (!/^1\d{10}$/.test(contact_phone)) return showToast('请输入正确的手机号', 'error');
      try {
        const r = await API.faults.create({ device_code, fault_category_l1, description, contact_person, contact_phone });
        // 关联配件
        if (_selectedParts.length > 0) {
          for (const sp of _selectedParts) {
            await API.post(`/api/parts/faults/${r.fault_no}/parts`, {
              part_id: sp.part_id, quantity: sp.quantity, note: description.slice(0, 100)
            });
          }
        }
        Modal.hide();
        showToast(`工单已创建：${r.fault_no}${_selectedParts.length > 0 ? '，关联配件' + _selectedParts.length + '项' : ''}`);
        navigateTo('faults');
      } catch(e) { showToast(e.message, 'error'); }
    }

    // ========== 消息通知 ==========
    async function renderNotifications(container) {
      container.innerHTML = `
        <div class="card">
          <div class="card-title">🔔 消息通知 <button class="btn btn-sm btn-outline" onclick="doReadAllNotifications()" style="float:right;">全部已读</button></div>
          <div id="notifListWrap"></div>
          <div class="pagination" id="notifPagination"></div>
        </div>
      `;
      reloadNotifications(1);
    }

    async function reloadNotifications(page = 1) {
      try {
        const r = await API.notifications.list({ page, page_size:20 });
        const wrap = document.getElementById('notifListWrap');
        if (!r.data || r.data.length === 0) { wrap.innerHTML = '<div class="empty-state"><p>暂无消息</p></div>'; return; }
        wrap.innerHTML = `
          <table class="data-table">
            <thead><tr><th>状态</th><th>标题</th><th>内容</th><th>类别</th><th>时间</th><th>操作</th></tr></thead>
            <tbody>
              ${r.data.map(n => `
                <tr style="${n.is_read?'':'font-weight:600;'}">
                  <td>${n.is_read?'<span style="color:var(--text-muted);">已读</span>':'<span style="color:var(--danger);">● 未读</span>'}</td>
                  <td>${n.title}</td>
                  <td>${(n.content||'').slice(0,50)}</td>
                  <td>${n.category}</td>
                  <td>${formatDate(n.created_at)}</td>
                  <td>${n.is_read?'-':`<button class="btn btn-sm btn-primary" onclick="doReadNotification(${n.id})">标为已读</button>`}${n.fault_no?`<button class="btn btn-sm btn-outline" onclick="navigateTo('faultDetail',{fault_no:'${n.fault_no}'});Modal.hide();" style="margin-left:4px;">查看工单</button>`:''}</td>
                </tr>
              `).join('')}
            </tbody>
          </table>
        `;
        renderPagination('notifPagination', r.total, r.page, r.page_size, reloadNotifications);
      } catch(e) {
        document.getElementById('notifListWrap').innerHTML = `<p style="color:var(--danger)">加载失败: ${e.message}</p>`;
      }
    }

    async function doReadNotification(id) {
      try {
        await API.notifications.read(id);
        reloadNotifications();
      } catch(e) { showToast(e.message, 'error'); }
    }

    async function doReadAllNotifications() {
      try {
        await API.notifications.readAll();
        showToast('已全部标为已读', 'success');
        reloadNotifications();
      } catch(e) { showToast(e.message, 'error'); }
    }

    function setTheme(t) {
      document.documentElement.setAttribute('data-theme', t);
      localStorage.setItem('theme', t);
      renderProfile(document.getElementById('pageContent'));
    }

    async function doChangePassword() {
      const old_password = document.getElementById('oldPass')?.value;
      const new_password = document.getElementById('newPass')?.value;
      if (!old_password || !new_password) return showToast('请填写新旧密码', 'error');
      if (new_password.length < 6) return showToast('新密码至少6位', 'error');
      try {
        await API.post('/api/auth/change-password', { old_password, new_password });
        showToast('密码修改成功，请重新登录', 'success');
        setTimeout(() => logout(), 1500);
      } catch(e) { showToast(e.message, 'error'); }
    }

    // ========== 分页 ==========
    function renderPagination(containerId, total, page, pageSize, callback) {
      const totalPages = Math.ceil(total / pageSize);
      if (totalPages <= 1) { document.getElementById(containerId).innerHTML = ''; return; }
      let html = '';
      html += `<button ${page<=1?'disabled':''} onclick="${callback.name}(${page-1})">‹ 上一页</button>`;
      const start = Math.max(1, page - 2);
      const end = Math.min(totalPages, page + 2);
      for (let i = start; i <= end; i++) {
        html += `<button class="${i===page?'active':''}" onclick="${callback.name}(${i})">${i}</button>`;
      }
      html += `<button ${page>=totalPages?'disabled':''} onclick="${callback.name}(${page+1})">下一页 ›</button>`;
      html += `<span style="margin-left:12px;font-size:12px;color:var(--text-muted);">共 ${total} 条</span>`;
      document.getElementById(containerId).innerHTML = html;
    }
